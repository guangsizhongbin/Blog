@Data
public class HumanBuilder {
    Human human = new Human();

    public HumanBuilder setIq(Integer iq) {
        Head head = human.getHead();
        head.setIq(iq);
        return this;
    }

    public HumanBuilder setHair(String hair) {
        Head head = human.getHead();
        head.setHair(hair);
        return this;
    }
}
