安装samba

1
2

sudo apt-get update
sudo apt-get intall samba samba-common-bin

配置smb.conf

/etc/samba/smb.conf

1
2
3
4
5
6

[share]
path = /home/pi
valid users = pi
browseable = yes
public = yes
writable = yes

重新运行smaba服务

sudo /etc/init.d/samba restart

添加pi用户为Samba用户

sudo smbpasswd -a pi
