@Configuration
@PropertySource("classpath:testaa.properties")
public class ConfigTest {

    @Value("${aaa.a1}")
    String a1;

    @Value("${aaa.a2}")
    String a2;

    @Value("${aaa.a3}")
    String a3;

    @Value("${aaa.a4}")
    String a4;

    @Bean
    public ConfigPojo getAaa() {
        ConfigPojo cp = new ConfigPojo();
        cp.a1 = a1;
        cp.a2 = a2;
        cp.a3 = a3;
        cp.a4 = a4;

        return cp;
    }
}
